    \begin{center}
        \begin{align*}
          \DCC(f) & \coloneqq \min\limits_{P \text{ for } f} \text{depth}(P) \\
          \RCC(f) & \coloneqq \min\limits_{\text{rand. } P \text{ for } f} \text{depth}(P)
        \end{align*}
