            \pause

            \pause
        \item Replace all $S_j$ by the $Z$.
    \end{itemize}

            \item $\Pr\limits_{W \sim \mathbf{U}_p}[\exists i, W \supseteq (S_i \setminus Z)] \ge 1 -
