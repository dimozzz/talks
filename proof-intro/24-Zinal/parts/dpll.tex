\begin{frame}{$\DPLL$ and Resolution}
    
    \begin{theorem}
        $\DPLL$ algoritm makes $t$ splitting on \alert{unsatisfiable} CNF formula
        $$\varphi \coloneqq \bigwedge\limits_i C_i$$
        $\Rightarrow$ there exists a resolution proof of $\varphi$ of size $2t$.
    \end{theorem}
