\def\hlinewd#1{%
\noalign{\ifnum0=`}\fi\hrule \@height #1 %
\futurelet\reserved@a\@xhline} 


\newcommand{\ordday}[1]{
	\multirow{3}{*}{\textbf{#1}}
    \orddaycontinue
}

}
