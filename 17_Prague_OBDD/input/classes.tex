    \else
	    \expandafter\newcounter{#2_help}
	\fi

    \restatable{#1}{#2}
    \expandafter\label{#2}
    \expandafter\ifnum\value{#2_help} = 0
    	\seepage[For proof see]{#2_app}
    \else
        \seepage{#2}
    \fi

}
